const getRecommendationBadge = (scenario: CompletionScenario) => {
  if (scenario.is_recommended) return { text: '최적', color: 'bg-emerald-600' };

  const aProb = scenario.grade_probabilities['A'] || 0;
  if (aProb >= 0.8) return { text: '양호', color: 'bg-blue-500' };
  if (aProb >= 0.6) return { text: '보통', color: 'bg-amber-500' };
  return { text: '주의', color: 'bg-red-500' };
};
